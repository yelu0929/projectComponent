import { Playground, Props } from 'docz';
import 'antd/dist/antd.css';
import LimitUpload from './LimitUpload';
import UploadType from '../enums/UploadType';
import UploadAcceptType from '../enums/UploadAcceptType';

# LimitUpload
可设置最大上传文件数量的组件。

当设置type时，但未设置accept时，会自动设置accept

## 示例

### 默认
<Playground>
  <LimitUpload action="http://www.baidu.com" />
</Playground>


### 限制上传数量
<Playground>
  <LimitUpload maxNumber={2} action="http://www.baidu.com"/>
</Playground>


### 图片上传
<Playground>
  <LimitUpload type={UploadType.IMAGE} action="http://www.baidu.com"/>
</Playground>

### office文档上传
<Playground>
  <LimitUpload accept={UploadAcceptType.OFFICE} />
</Playground>

### 自定义渲染方法
<Playground>
  <LimitUpload 
    action="http://www.baidu.com"
    chooserRender={()=> {
      return <span style={{borderLeft: '3px solid red', paddingLeft:10}}>点我上传</span>
      }}
  />
</Playground>


### 自定义传递给upload的属性

观察开发者工具，请求是PUT

<Playground>
  <LimitUpload 
    action="http://www.baidu.com"
    uploadProps={{
      method:'put'
    }}
  />
</Playground>
